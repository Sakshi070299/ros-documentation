  formatDownloadData() {
    this.bankings = [];
    let i = 1;
    let selectedNodes = this.gridApi.getSelectedNodes();
    let selectedData = selectedNodes.map((node) => node.data);
    console.log("selected data", selectedData);
    if (selectedData?.length > 0) {
      selectedData.forEach((x) => {
        let bankingObj = {};
        bankingObj["Sl. No"] = i++;
        bankingObj["Banking Date"] = x.bankingDate.toString().slice(0, 10);
        bankingObj["Banking Time"] = x.bankingTimeIndicator;
        bankingObj["No of CashUps"] = x.noOfcashUps;
        let deposit_dates_str = "";
        x.cashUpDates.map((x) => (deposit_dates_str += x.slice(0, 10) + ","));
        bankingObj["Deposit Dates"] = deposit_dates_str;
        bankingObj["Banking Amount"] = this.currSym + " " + x.bankingTotal;
        bankingObj["Banked Amount"] = this.currSym + " " + x.bankedTotal;
        bankingObj["Difference"] = this.currSym + " " + (x.bankingTotal - x.bankedTotal);
        bankingObj["Reason for Difference"] = x.reason;
        bankingObj["Giro Slip Number"] = x.giroSlipNumber;
        bankingObj["Sealed By"] = x.sealedBy;
        this.bankings.push(bankingObj);
      });
      console.log("allBanking data", this.allBanking);
      // this.downloadService.downloadExcel(this.bankings);
    } else {
      this.snackBar.open("Please make selection before download", "Close", {
        duration: 2000,
      });
    }
  }
