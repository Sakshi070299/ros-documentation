  upload(documentDetails: any) {
    this.progress.percentage = 0;
    this.uploadService.pushFileToStorage(this.url).subscribe((event) => {
      if (event.type === HttpEventType.UploadProgress) {
        this.progress.percentage = Math.round(
          (100 * event.loaded) / event.total
        );
        // console.log("Progress: ", this.progress);
      } else if (event instanceof HttpResponse) {
        const response: any = event.body;
        this.documents.push({
          document: documentDetails.docName,
          description: documentDetails.desp,
          documentUrl: response.fileDownloadUri,
          documentType: documentDetails.docType,
        });
        //this.documents = this.documents.slice();
        this.documents = this.documents.map((x) => {
          if (x.documentUrl.slice(0, 5) != "https")
            x.documentUrl = x.documentUrl.replace("http", "https");
          return x;
        });
      }
      this.empFormGroup.get("docname").reset();
      this.empFormGroup.get("doctype").reset();
      this.empFormGroup.get("Desp").reset();
      this.selectedFiles = undefined;
    });
