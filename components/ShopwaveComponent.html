import {
  Component,
  Input,
  OnInit,
  TemplateRef,
  ViewChild,
} from "@angular/core";
import { FormControl, FormGroup, Validators } from "@angular/forms";
import { MatDialogRef } from "@angular/material/dialog";
import { NbDialogService } from "@nebular/theme";
import { IntegrationFacadeService } from "../../facade/integration-facade.service";

@Component({
  selector: "ngx-shopwave",
  templateUrl: "./shopwave.component.html",
  styleUrls: ["./shopwave.component.scss"],
})
export class ShopwaveComponent implements OnInit {
  @Input() data: any = {};
  integrationForm: FormGroup;
  @ViewChild("Confirmation") Confirmation: TemplateRef<any>;
  token: any = null;

  constructor(
    private dialogRef: MatDialogRef<ShopwaveComponent>,
    private integrationFacadeService: IntegrationFacadeService,
    private dialogService: NbDialogService
  ) {}

  ngOnInit(): void {
    if (
      this.data.integrationCredentials != null &&
      this.data.integrationCredentials != ""
    ) {
      this.token = JSON.parse(this.data.integrationCredentials);
    }

    this.integrationForm = new FormGroup({
      storeId: new FormControl(
        this.token ? this.token.storeId : "",
        Validators.required
      ),
      accessToken: new FormControl(
        this.token ? this.token.accessToken : "",
        Validators.required
      ),
    });
  }

  confirm() {
    this.integrationFacadeService.editIntegration(
      this.data,
      this.data.id,
      true,
      encodeURIComponent(JSON.stringify(this.integrationForm.value))
    );
    this.dialogService.open(this.Confirmation);
    this.dialogRef.close();
  }

  deleteToken() {
    this.integrationFacadeService.editIntegration(
      this.data,
      this.data.id,
      false,
      null
    );
    this.dialogService.open(this.Confirmation);
    this.dialogRef.close();
  }
}
