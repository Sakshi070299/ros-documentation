import {
  Component,
  Input,
  OnDestroy,
  OnInit,
  TemplateRef,
  ViewChild,
} from "@angular/core";
import { MatDialogRef } from "@angular/material/dialog";
import { NbDialogService } from "@nebular/theme";
import { Subject } from "rxjs";
import { takeUntil } from "rxjs/operators";
import { IntegrationFacadeService } from "../../facade/integration-facade.service";
import { IntegrationService } from "../../integration.service";

@Component({
  selector: "ngx-foodics",
  templateUrl: "./foodics.component.html",
  styleUrls: ["./foodics.component.scss"],
})
export class FoodicsComponent implements OnInit, OnDestroy {
  @Input() data: any = {};
  private unsubscribe$ = new Subject<void>();
  haveFoodicsAccount: boolean = false;
  loggedInFoodics: boolean = false;
  isSubmitButtonDisabled: boolean = true;
  hasAccessToken: boolean;
  foodicsBranchData: any;
  rosRestaurantDetails: any;
  isBranchSelected: boolean = false;
  firstPage: Number = 1;
  foodicsBranchMetaData: any;
  selectedBranchId: string;
  rosClientId: string;
  rosRestaurnatId: string;
  accessToken: string;
  mappedBranches: string[] = [];
  isDataLoaded: boolean = false;

  columnsName: string[] = [
    "index",
    "name",
    "name_localized",
    "address",
    "reference",
    "selected",
  ];
  /** ChangesMade is triggered when there is change in mapping in UI */
  changesMade: boolean = false;
  initialMappedBranch: any;

  @ViewChild("Confirmation") Confirmation: TemplateRef<any>;
  @ViewChild("Unmapped") Unmapped: TemplateRef<any>;

  constructor(
    private integrationService: IntegrationService,
    private integrationFacadeService: IntegrationFacadeService,
    private dialogRef: MatDialogRef<FoodicsComponent>,
    private dialogService: NbDialogService
  ) {}

  ngOnInit(): void {
    this.rosClientId = JSON.parse(localStorage.getItem("User")).userProfile.id;
    this.rosRestaurnatId = localStorage.getItem("UserRestaurantId");

    this.integrationService
      .foodicsGetBranchesData(this.rosClientId)
      .pipe(takeUntil(this.unsubscribe$))
      .subscribe({
        next: (response: any) => {
          this.foodicsBranchData = response.body.foodicsBranches;
          this.getAllMappedBranches(response.body.mappedBranches);
          this.hasAccessToken = true;
          this.isDataLoaded = true;
        },
        error: (error) => {
          this.hasAccessToken = false;
          this.isDataLoaded = true;
        },
      });

    this.rosRestaurantDetails = JSON.parse(
      localStorage.getItem("UserRestaurant")
    );
  }

  private getAllMappedBranches(mappedData: any[]) {
    try{
    mappedData.forEach((data) => {
      this.mappedBranches.push(data.branchId);
      if (data.restaurantId === this.rosRestaurnatId) {
        this.selectedBranchId = data.branchId;
        this.initialMappedBranch = data.branchId;
      }
    });
  }
  catch(e){
    console.log(e)
  }
  }

  assert1() {
    this.haveFoodicsAccount = !this.haveFoodicsAccount;
    this.enableButton();
  }
  assert2() {
    this.loggedInFoodics = !this.loggedInFoodics;
    this.enableButton();
  }
  /** This function validates whether the confirm button is disabled */
  private enableButton() {
    this.isSubmitButtonDisabled = !(
      this.haveFoodicsAccount && this.loggedInFoodics
    );
  }

  /** Funtion to send a confirmation request to foodics to authorize ROS */
  confirm() {
    try{
    let randomString = this.generateRandomString();
    localStorage.setItem("FoodicsRandomString", randomString);
    window.location.href = `https://console.foodics.com/authorize?client_id=97e030d8-8a04-490f-8331-794940b1c249&state=${randomString}`;
    }
    catch(e){
      console.log(e);
      
    }
  }

  /** Mark the branch with green tick when it is been selected by user */
  branchSelected(data: any) {
    try {
      
   
    if (this.filterBranches(data)) return;
    if (
      this.initialMappedBranch != null ||
      this.initialMappedBranch != undefined
    )
      return;

    if (this.selectedBranchId) {
      if (this.selectedBranchId !== data) {
        this.selectedBranchId = data;
      } else {
        this.selectedBranchId = null;
        this.isBranchSelected = false;
      }
    } else {
      this.selectedBranchId = data;
      this.isBranchSelected = true;
    }

    if (this.initialMappedBranch != this.selectedBranchId) {
      this.changesMade = true;
    } else {
      this.changesMade = false;
    }
  } catch (error) {
      console.log(error);
      
  }
  }

  /** Method to send a request to Api to map a particular branch with restaurant */
  confirmBranch() {
    try {
      

    this.integrationService
      .foodicsAddMappingTemp(
        this.rosClientId,
        this.rosRestaurnatId,
        this.selectedBranchId
      )
      .pipe(takeUntil(this.unsubscribe$))
      .subscribe({
        next: (_res) => this.editIntegration(),
        error: (err) => console.log(err),
      });
    } catch (error) {
      console.log(error);
      
    }
  }

  removeBranch() {
    try{
    this.integrationService
      .foodicsRemoveMappingTemp(this.rosRestaurnatId)
      .pipe(takeUntil(this.unsubscribe$))
      .subscribe({
        next: (_res) => {
          this.isBranchSelected = false;
          this.editIntegration();
        },
        error: (err) => console.log(err),
      });
    } catch (error) {
      console.log(error);
      
    }
  }

  /** Calling this method to save branch id in integrationCredential */
  private editIntegration() {
    this.integrationFacadeService.editIntegration(
      this.data,
      this.data.id,
      this.isBranchSelected,
      this.isBranchSelected
        ? encodeURIComponent(
            JSON.stringify({
              branchId: this.selectedBranchId,
            })
          )
        : null
    );
    this.dialogService.open(
      this.isBranchSelected ? this.Confirmation : this.Unmapped
    );
    this.dialogRef.close();
  }

  /** Generating random string */
  private generateRandomString() {
    let lengthOfString = Math.floor(Math.random() * 5) + 6;
    let possibleCharacter = "ABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890";

    let randomString = "";
    for (var i = 0; i < lengthOfString; i++) {
      randomString += possibleCharacter.charAt(
        Math.floor(Math.random() * possibleCharacter.length)
      );
    }

    return randomString;
  }

  /** To check if branch id is already mapped */
  filterBranches(branchId: string): boolean {
    return this.mappedBranches.includes(branchId);
  }

  /** When component gets destroyed */
  ngOnDestroy(): void {
    this.unsubscribe$.next();
    this.unsubscribe$.complete();
  }
}
