  ngOnInit(): void {
    this.getResturantConfiguration();
  }

  getResturantConfiguration() {
    this.report_service
      .getResturantConfiguartionById(localStorage.getItem("UserRestaurantId"))
      .pipe(takeUntil(this.destroy$))
      .subscribe((data) => {
        this.thirdPartySources = data.thirdPartyConfig.thirdPartySource;
        this.pettyCashTypes = data.cashnPdqConfig.pettyCashTypes;
        this.pdqTypes = data.cashnPdqConfig.pdqTypes;
        this.tillTypes = data.cashnPdqConfig.tillTypes;
      });
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
    this.destroy$.unsubscribe();
  }

  @ViewChild("saveContent") saveContent: TemplateRef<any>;
  saveReport() {
    // this.deleteObj = row;
    this.startEndDateError = false;
    if (
      this.startDate.valid &&
      this.endDate.valid &&
      (this.showDiv.Epos ||
        this.showDiv.PettyCash ||
        this.showDiv.Pdq ||
        this.showDiv.ThirdPartyTakings ||
        this.showDiv.kpi ||
        this.showDiv.safeSummary)
    ) {
      this.dialogService.open(this.saveContent);
    } else {
      this.startEndDateError = true;
    }
  }

  onChecked(obj: string, isChecked: boolean) {
    this.showDiv[obj] = isChecked;
  }

  onEposChecked(obj: string, isChecked: boolean) {
    this.eposFeildsDTO[obj] = isChecked;
  }

  onKpiChecked(obj: string, isChecked: boolean) {
    this.kpiFeildsDTO[obj] = isChecked;
  }

  onPettyCashChecked(obj: string, isChecked: boolean) {
    if (isChecked) {
      this.pettyCash.push(obj);
    } else {
      const index = this.pettyCash.indexOf(obj, 0);
      this.pettyCash.splice(index, 1);
    }
  }

  onTillTypeChecked(obj: string, isChecked: boolean) {
    if (isChecked) {
      this.tillSystems.push(obj);
    } else {
      const index = this.tillSystems.indexOf(obj, 0);
      this.tillSystems.splice(index, 1);
    }
  }

  onPdqTypeChecked(obj: string, isChecked: boolean) {
    if (isChecked) {
      this.pdqs.push(obj);
    } else {
      const index = this.pettyCash.indexOf(obj, 0);
      this.pdqs.splice(index, 1);
    }
  }

  onThirdPatyChecked(obj: string, isChecked: boolean) {
    if (isChecked) {
      this.thirdParty.push(obj);
    } else {
      const index = this.pettyCash.indexOf(obj, 0);
      this.thirdParty.splice(index, 1);
    }
  }
  startDateV;

  dateValidations() {
    this.startEndDateError = false;
    this.startDateV = this.startDate.value;
    if (!this.startDate.valid && !this.endDate.valid) {
      this.startEndDateError = true;
    }
  }
  allreports() {
    if (this.reportName.valid) {
      let startDate = this.datepipe.transform(
        this.startDate.value,
        "yyyy-MM-dd"
      );
      let endDate = this.datepipe.transform(this.endDate.value, "yyyy-MM-dd");

      let report = {
        filename: this.reportName.value,
        createdBy: localStorage.getItem("AzureName"),
        fromDate: startDate + "T00:00:00.000Z",
        toDate: endDate + "T23:59:59.999Z",
        restaurantId: localStorage.getItem("UserRestaurantId"),
        eposfieldsDto: this.eposFeildsDTO,
        cashPDQFieldDto: {
          pettyCash: this.pettyCash,
          tillSystems: this.tillSystems,
          pdqs: this.pdqs,
          wageAdvances: false,
        },
        thirdParty: this.thirdParty,
        kpiFields: this.kpiFeildsDTO,
        safeSummary: this.showDiv.kpi,
      };
      this.report_service.createNewReport(report).subscribe((data) => {
        this.router.navigateByUrl("/accounting/report/home");
      });
    }
  }
}
