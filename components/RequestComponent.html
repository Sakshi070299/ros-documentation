  calculateTotalHours(punchIn, punchOut) {
    const inTimestamp = new Date(punchIn);
    const inHours = inTimestamp.getUTCHours();
    const inMinutes = inTimestamp.getUTCMinutes();

    const outTimestamp = new Date(punchOut);
    const outHours = outTimestamp.getUTCHours();
    const outMinutes = outTimestamp.getUTCMinutes();

    const totalMinutes = (outHours * 60 + outMinutes) - (inHours * 60 + inMinutes);

    const hours = Math.floor(totalMinutes / 60);
    const minutes = totalMinutes % 60;

    const timeComponents = [];

    if (hours > 0) {
      timeComponents.push(hours + " H");
    }

    if (minutes > 0) {
      timeComponents.push(minutes + " M");
    }

    const formattedTime = timeComponents.join(" ");

    return formattedTime;
  }

  onClickRejectLogic() {
    if (this.allCheckedShiftReq.length !== 0) {

      let selectedRow = this.allCheckedShiftReq.filter(data => {
        return data.requestType != 'PunchIn_OutRequest';
      })

      if (selectedRow.length > 0 && selectedRow.length != this.allCheckedShiftReq.length) {
        this.shiftconfirmation("Can't Reject both request at a time!!");
        return
      }

      if (selectedRow.length > 0 && selectedRow.length == this.allCheckedShiftReq.length) {

        // call shiftRequest reject
        let something = selectedRow.filter(data => {
          return data.approvalStatus == 'Pending';
        })
        this.onClickRejectShift(something);
        // this.editBulkStatus1(something);


      }

      if (selectedRow.length == 0) {
        // call punchinOut reject
        let PendingReq = this.allCheckedShiftReq.filter(data => {
          return data.approvalStatus == 'Pending';
        })

        if (PendingReq.length != 0) {
         let Pending = PendingReq.filter(data => {
            return (data.punchIn != null) || (data.punchOut != null);
          })
        Pending.length != 0?
            this.onClickRejectPunchInOutRequest(Pending)
         :
            this.openErrorPopup("We Can't Reject the shift punchIn/Out is Null !!");
          
        }
        else {
          this.openErrorPopup("Shift has been already Approved/Rejected!!");

        }


      }

    }
    else {
      this.openErrorPopup("No records has been selected!!");
    }
  }
