  totalEpos() {
    let res = { sales: 0.0, tax: 0.0, tips: 0.0, total: 0.0 };
    let val = this.cashFG.value;
    res.sales +=
      Number(val.food) +
      Number(val.drinks) -
      Number(val.discountTotal) +
      Number(val.others);
    res.tax += Number(val.Vat);
    res.tips += Number(val.CCtip) + Number(val.Scharges);
    res.total +=
      Number(res.sales) +
      Number(res.tax) +
      Number(res.tips) +
      Number(res.total);
    return res;
  }

  totalCashPdq() {
    let res = { petty_cash: 0.0, till: 0.0, pdq: 0.0, wage_adv: 0.0 };
    let val = this.cashFG.value;
    res.petty_cash += val.petty_cash_array.reduce(
      (t, x) => Number(t) + Number(x.petty_cash_name),
      0
    );
    res.till += val.tills_array.reduce(
      (t, x) => Number(t) + Number(x.till_name),
      0
    );
    val.pdqs_array.forEach((x) => {
      x.pdqCard.forEach((y) => {
        res.pdq += Number(y.pdq_amount);
      });
    });
    res.wage_adv += val.wage_advances_array.reduce(
      (t, x) => Number(t) + Number(x.amount),
      0
    );

    return res;
  }

      (sum, x) => Number(sum) + Number(x.amount),
      0
    );

    res.dis_breakdown += val.discount_breakdown_array.reduce(
      (sum, x) => Number(sum) + Number(x.amount),
      0
    );
