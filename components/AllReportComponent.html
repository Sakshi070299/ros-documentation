  delReports() {
    let selectedNodes = this.gridApi.getSelectedNodes();
    console.log("selectedNodes", selectedNodes);
    let selectedData = selectedNodes.map((node) => node.data);
    console.log("selected data", selectedData);
    this.i = selectedData.length;
    if (selectedData?.length > 0) {
      selectedData.forEach((x) => {
        this.deleteMultipleReportData(x);
        this.getAllReports();
      });
    }
    else {
      this.snackBar.open("Please make selection before delete", "Close", {
        duration: 2000,
      });
    }
  }

  deleteMultipleReportData(d: ReportsModel) {
    this.reportService.deleteReportById(d.id, d.filename).subscribe((data) => {
      this.delMsg = data;
      this.i--;
      if (this.i == 0)
        this.getAllReports();
      console.log("Delete Message: ", this.delMsg);
    });
  }

  deleteReportData(d: ReportsModel) {
    this.reportService.deleteReportById(d.id, d.filename).subscribe(data => this.getAllReports())
    if (this.reportSelected.includes(d.filename)) {
      const index = this.reportSelected.indexOf(d.filename, 0)
      this.reportSelected.splice(index, 1)
    }
  }

  dateFormatter(date: Date) {
    var formatted_date = this.datePipe.transform(date, "dd MMM yyyy");
    return formatted_date;
  }

async  downloadReport() {
    if (this.reportSelected.length != 0) {
      for (let reportName of this.reportSelected) {
        let url = `${environment.backendUrl}account/report/${reportName}`;
        let res=  await fetch(url)
        if(res.status===200){
        const link = document.createElement("a");
        link.setAttribute("target", "_blank");
        link.setAttribute("href", url);
        link.setAttribute("download", url);
        document.body.appendChild(link);
        link.click();
        }
        else{
       

          this.dialogService.open(this.notexist);
          
            }
      }
    }
  }

  onRowSelected(obj) {
    if (this.reportSelected.includes(obj.data.filename)) {
      const index = this.reportSelected.indexOf(obj.data.filename, 0)
      this.reportSelected.splice(index, 1)
    }
    else {
      this.reportSelected.push(obj.data.filename)
    }
  }
