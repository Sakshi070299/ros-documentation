    let dates = [];

    // console.log("Dates 2", this.selectedDeposits);

    return dates;
  }

  addSelectedDeposit(d) {
    let obj = JSON.parse(d);
    // console.log(obj);
    let selDate = new Date(obj.cashUpDate);
    if (
      obj &&
      !this.selectedDeposits.find(
        (x) =>
          new Date(x.cashUpDate).toISOString() ==
          new Date(obj.cashUpDate).toISOString()
      )
    ) {
      this.selectedDeposits.push(obj);
      // this.Cashup_sheet_dates.push(obj.cashUpDate);
      this.logger.log("add date", this.selectedDeposits);
      // console.log("Sel Dep Dates", this.getSelectedCashupDates());
      // this.logger.log("add cashup", this.Cashup_sheet_dates);
    }
    this.OnRefreshingdata();
  }

  OnRefreshingdata() {
    var sum = 0;
    for (let bankingtotal of this.selectedDeposits) {
      sum =
        sum +
        bankingtotal.cashnPdq.tillSystems.reduce((sum, x) => sum + x.amount, 0);
    }
    this.BankingTotalSum = parseFloat(sum.toFixed(2));
    this.logger.log("bankingTotalSum", sum);
    this.bankingForm.patchValue({
      banking_total: this.BankingTotalSum,
    });
  }
  removeDeposit(d: CashUp) {
    this.selectedDeposits.splice(
      this.selectedDeposits.findIndex((x) => x.cashUpDate == d.cashUpDate),
      1
    );
    this.logger.log("date remove", this.selectedDeposits);
    // this.Cashup_sheet_dates.splice(

    //   1
    // );
    // this.logger.log("date remove", this.Cashup_sheet_dates);
    this.OnRefreshingdata();
  }
