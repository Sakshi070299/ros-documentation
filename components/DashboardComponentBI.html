import { Component, OnInit, ViewChild,ElementRef, } from '@angular/core';
import { IReportEmbedConfiguration, models, service, Embed } from 'powerbi-client';
import * as pbi from 'powerbi-client'; 
// import { powerbi } from 'powerbi-client'
// declare var powerbi: any;
import { AuthServiceService } from "../auth/service/auth-service.service";


@Component({
    selector: "app-dashboard",
    templateUrl: "./dashboard-bi.component.html"
  })
  
  export class DashboardComponentBI implements OnInit{
    public screenHeight:number;
    token:any;
    restaurantDetails:any
    restaurants=[];
  

      // @ViewChild('reportContainer') reportContainer: ElementRef; 
  
      constructor(private authServiceService: AuthServiceService,) { }
      ngOnInit() {  
        let email= localStorage.getItem('AzureEmail')
        //   this.restaurants.push(this.restaurantDetails.id)
         this.authServiceService.getUserRestaurants(email).subscribe(restaurant => {
          console.log(restaurant,"===restaurant");
          restaurant.restaurants.map((item)=>{
            this.restaurants.push(item?.id)
          })
        })
          this.screenHeight = (window.screen.height);  
       this.getToken(); 
      //  console.log("====hiii",this.restaurantDetails,this.restaurants);

        // this.showReport();  
   }  

  //  ngAfterViewInit() {
  //   this.showReport();
  // }

  async getToken(){
   await this.authServiceService.getPowerBiDashboard().subscribe(token => {
      // Handle the token here     
      this.token=token  
      this.showReport(this.token);  
      console.log(this.token,"=====this.token")
    });
   }

   showReport(token) {  
    // Embed URL    
    let embedUrl = token?.embedReports[0]?.embedUrl 
    // let embedReportId = POWERBI.REPORT_ID;  
    const filter1 =
    {
        $schema: "http://powerbi.com/product/schema#basic",
        target: { table: "public restaurant (2)", column: "restaurant_id" },
        operator: "In",
        //blow values are restaurant_id, those should come dynamically from the local storage as per login user
        // values: ["1fd6c987-7182-4edd-ae23-6ef94c7b4d6f","3671e3fc-ce51-4b3a-b255-40068e63eab7"]
        values: this.restaurants
    };
console.log(token?.embedToken,"=token?.embedToken");

    let config = {  
        type: 'report',  
        accessToken: token?.embedToken,  
        embedUrl: embedUrl,  
        tokenType: models.TokenType.Embed,
        // id: 'c0853ee6-680d-406d-ae51-5512fb73923',  
         filters: [filter1], 
        // filters: ["1fd6c987-7182-4edd-ae23-6ef94c7b4d6f","3671e3fc-ce51-4b3a-b255-40068e63eab7"], 
    };  
    
    const reportsContainer = <HTMLElement>document.getElementById(
      'reportsContainer'
    );
    let powerbi = new pbi.service.Service(pbi.factories.hpmFactory, pbi.factories.wpmpFactory, pbi.factories.routerFactory);  
    
      let report = powerbi.embed(reportsContainer, config);  
      // Your embed code here
      report.off("loaded");  
      report.on("loaded", () => {  
          console.log("Loaded");  
      });  
      report.on("error", () => {  
          this.getToken();  
      });  
}  
}
