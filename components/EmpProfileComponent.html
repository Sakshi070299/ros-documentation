import { Component, OnInit, TemplateRef, ViewChild } from "@angular/core";
import { FormControl } from "@angular/forms";
import { Title } from "@angular/platform-browser";
import { ActivatedRoute, Router } from "@angular/router";
import { NbDialogService } from "@nebular/theme";
import { th } from "date-fns/locale";
import { AgGridCustomService } from "../../../shared/services/ag-grid-custom.service";
import { AuthFacadeService } from "../../auth/facade/auth-facade.service";
import { ViewDocumentActionCellComponent } from "../employees/view-employee/document-action-cell/view-document-action-cell.component";
import { EmployeeFacadeService } from "../fascade/employee-facade.service";
import { MyprofileActionCellRendererComponent } from "./myprofile-action-cell-renderer/myprofile-action-cell-renderer.component";
import { MyprofileShiftActionCellComponent } from "./myprofile-shift-action-cell/myprofile-shift-action-cell.component";

@Component({
  selector: "ngx-emp-profile",
  templateUrl: "./emp-profile.component.html",
  styleUrls: ["./emp-profile.component.scss"],
})
export class EmpProfileComponent implements OnInit {
  empProfileSubjectList$ = this.employeeFacade.empProfileSubjectList$;
  empshiftCalendarSubject$ = this.employeeFacade.empshiftCalendarList$;
  empFormTab = "Basic";

  gridApi: any;
  gridColumnApi: any;
  empid; // consider 0 as employee ID o
  empobj: any;
  empShiftobj: any;
  rosInfo= null;
  constructor(
    private router: Router,
    private employeeFacade: EmployeeFacadeService,
    private route: ActivatedRoute,
    private dialogService: NbDialogService,
    private authFacade: AuthFacadeService,
    private agGridService: AgGridCustomService,
    private title: Title
  ) {
    this.title.setTitle("ROS - User Profile");
    // if (this.router.getCurrentNavigation().extras.state) {
    //   this.rosInfo = this.router.getCurrentNavigation().extras.state.isROSTeam;
    //   console.log("sssssssssssssssssssssssssssssssssssssssssssss")
    //   console.log("from rosTeamssssssssssssssssssss", this.rosInfo);
    // }
  }

  allEmployee: any;
  documentList: any;
  UserName: string;
  BankName;
  BranchAccount;
  AccHolder;
  SortCode;
  AccNumber;
  EmpType;
  salary;
  totalHolidays;
  ReportingToID;
  ReportingEmpName;
  AzureEmail;
  location;
  resLocation;
  floatLabelControl4;
  ngOnInit(): void {
    this.UserName = this.authFacade.getAzureEmail();
    //this.UserName = "ganesh@gmail.com";
    console.log("AzureEmail", this.UserName);
    this.employeeFacade.employeeProfile(this.UserName);
    this.empProfileSubjectList$.pipe().subscribe((Data) => {
      this.empobj = Data;
      console.log("empobj:", this.empobj);
      this.ReportingToID = this.empobj.reportingEmpId;
      if (this.ReportingToID === this.empobj.empId) {
        //this.ReportingEmpName = this.allEmployee.firstName+" "+this.allEmployee.lastName;
        this.ReportingEmpName = "Self";
      } else {
        this.reportTo();
      }
      this.floatLabelControl4 = new FormControl(this.empobj.location);
      const empObjImg = this.empobj?.profilePic;
      this.BankName = this.empobj?.employeeBankInfo?.[0].bankName;
      this.BranchAccount = this.empobj?.employeeBankInfo?.[0].branchAccount;
      this.AccHolder = this.empobj?.employeeBankInfo?.[0].holderName;
      this.SortCode = this.empobj?.employeeBankInfo?.[0].sortCode;
      this.AccNumber = this.empobj?.employeeBankInfo?.[0].accNo;
      this.EmpType =
        this.empobj?.employeeBankInfo?.[0].salaryInfo?.employeeType;
      this.salary = this.empobj?.employeeBankInfo?.[0].salaryInfo?.salary;
      this.totalHolidays= this.empobj?.employeeBankInfo?.[0].salaryInfo?.totalHolidays;

    });
    //for location
