import { Component, OnInit, TemplateRef, ViewChild,Pipe } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { first } from 'rxjs/operators';
import { employeeDocument } from '../../model/employeeDocument.model';
import { EmployeeService } from '../../services/employee.service';
import { CustomValidators } from '../CustomValidators';
import { pipe } from 'rxjs';
import { NbDialogService } from '@nebular/theme';
import { connectableObservableDescriptor } from 'rxjs/internal/observable/ConnectableObservable';

@Component({
  selector: 'ngx-change-emp-pin',
  templateUrl: './change-emp-pin.component.html',
  styleUrls: ['./change-emp-pin.component.scss']
})
export class ChangeEmpPinComponent implements OnInit {
  @ViewChild("secondDialog") secondDialog: TemplateRef<any>;
  @ViewChild("successDialog") successDialog: TemplateRef<any>;
  @ViewChild("backDialog") backDialog: TemplateRef<any>;


  dialogMessage: String;
  buttonMessage: String;
  urlPath: String;
  editFlag: boolean;
  newPin: boolean = false;
  oldPin: boolean = false;
  empId: string;
  confirmPin: boolean = false;


  clientForm = new FormGroup({
    oldPin: new FormControl('', [Validators.required, Validators.minLength(6), Validators.maxLength(6),Validators.pattern("^[0-9]*$")]),
    newPin: new FormControl('', [Validators.required, Validators.minLength(6), Validators.maxLength(6),Validators.pattern("^[0-9]*$")]),
    confirmPin: new FormControl('', [Validators.required, Validators.minLength(6), Validators.maxLength(6),Validators.pattern("^[0-9]*$")]),
  },
    CustomValidators.mustMatch('newPin', 'confirmPin')
  );

  formBuilder: any;

  submitted = false;

  keyPressNumbers(event) {
    var charCode = (event.which) ? event.which : event.keyCode;
    // Only Numbers 0-9
    if ((charCode < 48 || charCode > 57)) {
      event.preventDefault();
      return false;
    } else {
      return true;
    }
  }


  constructor(private router: Router,   private dialogService: NbDialogService,public dialog: MatDialog, private empService: EmployeeService) {
 
    this.empId = this.router.getCurrentNavigation().extras.state['empId'];
   
  }




  ngOnInit() {
    const id = localStorage.getItem("empId");

  }

  public togglePasswordVisibility(): void {
    this.oldPin = !this.oldPin;
  }

  public togglePasswordVisibility1(): void {
    this.newPin = !this.newPin;
  }

  public togglePasswordVisibility2(): void {
    this.confirmPin = !this.confirmPin;
  }


  async onSubmit() {
    this.openDialog();

  }

  get f() {
    return this.clientForm.controls;
  }


  openDialog() {
    this.dialog.open(this.secondDialog, { disableClose: true });
  }

  openNavigationDialog() {
    this.dialog.open(this.successDialog, { disableClose: true });
