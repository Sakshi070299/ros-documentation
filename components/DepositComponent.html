import { Component, OnInit } from "@angular/core";
import { Router } from "@angular/router";
import { Subject } from "rxjs";
import { map, takeUntil } from "rxjs/operators";
import { CashupFacadeService } from "../facade/cashup-facade.service";
import { CashUp } from "../model/cashup.model";
import { Title } from "@angular/platform-browser";
import { DepositService } from "../service/deposit.service";
import { NGXLogger } from "ngx-logger";
import { DepositFacadeService } from "../facade/deposit-facade.service";
import { Banking } from "../model/deposit";
import { isSameDay, subDays } from "date-fns";

@Component({
  selector: "ngx-deposit",
  templateUrl: "./deposit.component.html",
  styleUrls: ["./deposit.component.scss"],
})
export class DepositComponent implements OnInit {
  selectedCard = "PENDING DEPOSITS";
  allCashup: CashUp[] = [];
  cashupList$ = this.cashupFacade.cashupList$;
  cashList$ = this.cashupFacade.cashList$;
  BankingList$ = this.bankingFacade.BankingList$;

  constructor(
    private depositService: DepositService,
    private cashupFacade: CashupFacadeService,
    private router: Router,
    private title: Title,
    private logger: NGXLogger,
    private bankingFacade: DepositFacadeService
  ) {
    this.title.setTitle("ROS - Deposit");
    this.prevDate.setDate(this.myDate.getDate() - 1);
  }

  private readonly destroy$ = new Subject<void>();
  cashupDraft$ = this.cashupFacade.cashupList$.pipe(
    takeUntil(this.destroy$),
    map((data) => data.filter((x) => x.cashUpStatus == "DRAFT")?.length)
  );
  all_;
  myDate = new Date();
  prevDate = new Date();
  deposits_filter = "M";
  allDeposit;
  allBanking: Banking[] = [];
  TotalDeposit;
  TotalBanking;

  ngOnInit(): void {
    this.cashList$.pipe(takeUntil(this.destroy$)).subscribe((cashupData) => {
      this.allDeposit = cashupData
      // Object.assign(
      //   [],
      //   cashupData.filter(
      //     (x) =>
      //       x.cashUpStatus === "PUBLISHED" &&
      //       !(
      //         x.cashnPdq.tillSystems.reduce((sum, x) => sum + x.amount, 0) === 0
      //       )
      //   )
      // );
      this.logger.log("Deposit List is updated", this.allDeposit);
    });
    // this.allDeposit = this.depositService.getAllDeposits();
    this.refreshBankingData();
    this.cashupList$
      .pipe(takeUntil(this.destroy$))
      .subscribe((cashupData: CashUp[]) => {
        this.allCashup = cashupData;
      });
    this.BankingList$.pipe().subscribe((data) => {
      this.allBanking = Object.assign([], data);
      console.log("Banking List is updated");
    });
    var sum = 0;
    for (let Depositvalue of this.allDeposit) {
      sum = sum + Depositvalue.cash_taking;
    }
    this.TotalDeposit = sum;
    var sum1 = 0;
    for (let Bankingvalue of this.allBanking) {
      sum1 = sum1 + Bankingvalue.bankedTotal;
    }
    this.TotalBanking = sum1;
  }
  RestaurantId;

  refreshBankingData() {
    this.RestaurantId = localStorage.getItem("UserRestaurantId")
    this.bankingFacade.load();
  }

  setSelectedCard(s: string) {
    this.selectedCard = s;
  }

  reset_deposits(s: string) {
    this.deposits_filter = s;
  }

  addNew() {
    this.router.navigateByUrl("/accounting/cashup/new");
  }

  filterbyDraft() {
    this.allCashup = this.allCashup.filter((x) => x.cashUpStatus == "Draft");
    // this.logger.log("filterByDraft",this.filterbyDraft())
  }

  ngOnDestroy(): void {
    this.destroy$.next();
    this.destroy$.complete();
  }
  ranges;
  range;
  filterByDateRange(ranges) {
    this.ranges = Object.assign({}, ranges);
    this.bankingFacade.filterByDateRange(ranges);
    this.cashupFacade.filterByDateRange(ranges);
  }
  getTotalDrafts() {
    return this.allCashup.reduce((sum, x) => {
      return sum + this.cashupFacade.getCashupTotals(x).cash;
    }, 0);
  }
  getTotalPendingdeposit() {
    return this.allDeposit.reduce((sum, x) => {
      return sum + this.cashupFacade.getCashupTotals(x).cash;
    }, 0);
  }
  getTotalBanking() {
    return this.allBanking.reduce((sum, x) => {
      return sum + x.bankedTotal;
    }, 0);
  }
  getTotalSummary() {
    let result = {
      yesterdayTotal: 0,
      todayTotal: 0,
      yesterdayPercent: 0,
      todayPercent: 0,
      showAddNew: true,
    };
    let todaySheets = this.allCashup.filter((x) =>
      isSameDay(new Date(x.cashUpDate), new Date())
    );
    let yesterdayDate = subDays(new Date(), 1);
    let yesterdaySheets = this.allCashup.filter((x) =>
      isSameDay(new Date(x.cashUpDate), yesterdayDate)
    );

    result.todayTotal = todaySheets.reduce((sum, x) => {
      return sum + this.cashupFacade.getCashupTotals(x).cash;
    }, 0);

    result.yesterdayTotal = yesterdaySheets.reduce((sum, x) => {
      return sum + this.cashupFacade.getCashupTotals(x).cash;
    }, 0);

    let dbYesterdayDate = subDays(new Date(), 2);
    let dbYesterdaySheets = this.allCashup.filter((x) =>
      isSameDay(new Date(x.cashUpDate), dbYesterdayDate)
    );

    let dbYesterdayTotal = dbYesterdaySheets.reduce((sum, x) => {
      return sum + this.cashupFacade.getCashupTotals(x).cash;
    }, 0);

    result.todayPercent =
      (result.todayTotal / (result.todayTotal + result.yesterdayTotal)) * 100 ||
      0;
    result.yesterdayPercent =
      (result.yesterdayTotal / (dbYesterdayTotal + result.yesterdayTotal)) *
        100 || 0;
    result.showAddNew = todaySheets.length != 2;

    return result;
  }
}
