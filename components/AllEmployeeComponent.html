 async onFileSelected(event: any) {
    const fileInput = event.target;
    if (fileInput.files && fileInput.files.length > 0) {
      const selectedFile = fileInput.files[0];
      this.showLoader = true;
      // Create FormData object to send as multipart/form-data
      const formData = new FormData();
      formData.append("file", selectedFile);
      if (selectedFile.name.endsWith(".xlsx")) {
        let access_token = await this.azureService.getToken();
        console.log(access_token,"==access_token");
        
        // Make the HTTP POST request
        this.employeeService.uploadFile(formData,access_token.access_token).subscribe(
          (response) => {
            this.successCashup();
            this.employeeReloadService.triggerReload();
            this.getAllReportingEmployee();

            console.log("API call successful:", response);
          },
          (error) => {
            this.getAllReportingEmployee();
            this.openErrorPopup(error);
            console.error("API call failed:", error);
          },
          () => {
            // Hide loader on completion (success or error)
            this.showLoader = false;
          }
        );
      } else {
        console.warn("Selected file is not a .xlsx file.");
        // You can display a message to the user or handle the invalid file in another way.
      }
    }
  }
