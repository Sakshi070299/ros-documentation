  getTotalSummary() {
    let todaySheets = this.allCashup.filter((x) =>
      isSameDay(new Date(x.cashUpDate), new Date())
    );
    let yesterdayDate = subDays(new Date(), 1);
    let yesterdaySheets = this.allCashup.filter((x) =>
      isSameDay(new Date(x.cashUpDate), yesterdayDate)
    );

    this.result.todayTotal = todaySheets.reduce((sum, x) => {
      return (
        sum +
        this.authServiceService.getCashupTotals(x).cash +
        this.authServiceService.getCashupTotals(x).petty
      );
    }, 0);

    this.result.yesterdayTotal = yesterdaySheets.reduce((sum, x) => {
      return (
        sum +
        this.authServiceService.getCashupTotals(x).cash +
        this.authServiceService.getCashupTotals(x).petty
      );
    }, 0);

    let dbYesterdayDate = subDays(new Date(), 2);
    let dbYesterdaySheets = this.allCashup.filter((x) =>
      isSameDay(new Date(x.cashUpDate), dbYesterdayDate)
    );

    let dbYesterdayTotal = dbYesterdaySheets.reduce((sum, x) => {
      return sum + this.authServiceService.getCashupTotals(x).cash;
    }, 0);

    this.result.todayPercent =
      ((this.result.todayTotal - this.result.yesterdayTotal) /
        this.result.todayTotal) *
        100 || 0;
    if (this.result.yesterdayTotal == 0) this.result.yesterdayPercent = -100;
    else
      this.result.yesterdayPercent =
        ((this.result.yesterdayTotal - dbYesterdayTotal) /
          this.result.yesterdayTotal) *
          100 || 0;

    this.result.showAddNew = todaySheets.length != 2;
    this.result.todayPercent.toString() == "-Infinity"
      ? (this.result.isTodayPercentage = false)
      : (this.result.isTodayPercentage = true);
    this.result.yesterdayPercent.toString() == "-Infinity"
      ? (this.result.isYesterdayPercentage = false)
      : (this.result.isYesterdayPercentage = true);
    return this.result;
  }

  calculateWTD() {
    const startOfWeek = moment().startOf('week').add(1, 'day'); // Monday of the current week
    const prevWeekStart = moment().subtract(1, 'week').startOf('week').add(1, 'day');
    const endOfWeek = moment().endOf('week').startOf('day').add(1, 'days');
    const endOfPreviousWeek = moment(prevWeekStart).endOf('week').endOf('day').add(1, 'day');


    const wtdSales = this.allCashup.filter((sale) => {
      const saleDate = moment(sale.cashUpDate); // Assuming sale.cashUpDate is a valid date string
      return saleDate.isBetween(startOfWeek, endOfWeek, null, '[]');
    });
   
    const prevWeekSales = this.allCashup.filter((sale) => {
      const saleDate = moment(sale.cashUpDate); 
      return saleDate.isBetween(prevWeekStart, endOfPreviousWeek, null, '[]');
    });
  
    this.wtdNetSales = wtdSales.reduce(
      (total, sale) => total + (sale?.epostotal - (sale?.sales[0]?.taxInfo[0]?.amount || 0) || 0),
      0
    );
  
    const prevWeekNetSales = prevWeekSales.reduce(
      (total, sale) => total + (sale?.epostotal - (sale?.sales[0]?.taxInfo[0]?.amount || 0) ?? 0) ,
      0
    );
    this.diffFromPrevWeek = this.wtdNetSales - prevWeekNetSales;

    const percentageChange = ((Math.abs(this.diffFromPrevWeek) / prevWeekNetSales) * 100);
    this.percentageDiffFromPrevWeek  = isNaN(percentageChange) ? 0 : Math.floor(percentageChange);
    

  }

      const cashUpDate = new Date(sale.cashUpDate);
      return cashUpDate >= this.start_date_cashup && cashUpDate <= this.end_date_cashup;
    });
  
    for (const entry of cashUpData) {
      for (const sale of entry.sales) {
        this.sums["Food"] += sale.foodPayment || 0;
        this.sums["Drink"] += sale.drinksPayment || 0;
        this.sums["Others"] += sale.otherPayment || 0;
        this.sums["Service Charge"] += sale.serviceCharges || 0;
        this.sums["Tips"] += sale.creditCardTip || 0;
        this.sums["Discount"] += sale.discount || 0;
        for (const taxInfo of sale.taxInfo) {
          this.sums["Tax"] += taxInfo.amount || 0;
        }
      }
    } 
    this.setChart();
  }
  
  getAllRestaurants(){
  this.authFacade.getUserRestaurants(localStorage.getItem("AzureEmail")).subscribe((x) => {
  
    let transformedRestaurants = x['restaurants'].reduce((acc, restaurant) => {
      const existingRestaurant = acc.find((r) => r.id === restaurant.id);
    
      if (existingRestaurant) {
        existingRestaurant.owners.push(restaurant.ownerName);
      } else {
        acc.push({
          id: restaurant.id,
          name: restaurant.name,
          owners: [restaurant.ownerName]
        });
      }
    
      return acc;
    }, []);
    this.restaurants  = transformedRestaurants;
  
  });
