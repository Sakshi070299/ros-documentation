  processEventData(events) {
    // Use an object to accumulate data based on date
    const dateMap = {};

    // Iterate through the events array
    events.forEach((event) => {
      const startDate = new Date(event.start).toLocaleDateString();

      // If the date is not in the map, add it with the current event
      if (!dateMap[startDate]) {
        dateMap[startDate] = { ...event };
      } else {
        // If the date is already in the map, update the title
        dateMap[startDate].title += event.title;
      }
    });

    // Convert the values of the map back to an array
    const result = Object.values(dateMap);

    return result;
  }
