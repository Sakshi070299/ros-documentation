  // Calculate the total number of pages
  getTotalPages(): number {
    this.totalItems = this.allMappedUser.length;
    // console.log(this.totalItems);

    return Math.ceil(this.totalItems / this.itemsPerPage);
  }

  // Get an array of page numbers for pagination
  getPageNumbers(): number[] {
    const pageCount = this.getTotalPages();
    const visiblePages = 5; // Number of visible pages (adjust as needed)

    let startPage: number, endPage: number;
    if (pageCount <= visiblePages) {
      startPage = 1;
      endPage = pageCount;
    } else {
      const half = Math.floor(visiblePages / 2);
      if (this.currentPage <= half) {
        startPage = 1;
        endPage = visiblePages;
      } else if (this.currentPage + half >= pageCount) {
        startPage = pageCount - visiblePages + 1;
        endPage = pageCount;
      } else {
        startPage = this.currentPage - half;
        endPage = this.currentPage + half;
      }
    }
    return Array.from(
      { length: endPage - startPage + 1 },
      (_, i) => startPage + i
    );
  }

  onPageChange(pageNumber: number) {
    this.currentPage = pageNumber;
    // Perform operations when page changes, e.g., fetch data for the new page
  }

  onFirstPage() {
    this.onPageChange(1);
  }

  onLastPage() {
    this.onPageChange(this.getTotalPages());
  }

  onPrevPage() {
    if (this.currentPage > 1) {
      this.onPageChange(this.currentPage - 1);
    }
  }

  onNextPage() {
    if (this.currentPage < this.getTotalPages()) {
      this.onPageChange(this.currentPage + 1);
    }
  }
