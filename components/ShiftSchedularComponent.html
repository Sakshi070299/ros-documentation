calculatePayroll(data) {
  let payroll = 0;
  const dataDateStr = data.date.toLocaleDateString(); // Convert data.date to date string

  // Check if it's a weekend (Saturday or Sunday)
  if (data.day === 'Saturday' || data.day === 'Sunday') {
    payroll = 0;
  } else {
    // It's a weekday (Monday to Friday)
    
    // Calculate the sum of hourly employee payroll for the specific date
    const hourlyPayroll = this.tempEmps
      .filter(emp => emp.employeeType === "HOURLYEMPLOYEE" && emp.shifts.some(shift => {
        const shiftDateStr = new Date(shift.date).toLocaleDateString(); // Convert shift.date to date string
        return shiftDateStr === dataDateStr;
      }))
      .reduce((sum, emp) => {
        const dayPayroll = emp.shifts
          .filter(shift => {
            const shiftDateStr = new Date(shift.date).toLocaleDateString(); // Convert shift.date to date string
            return shiftDateStr === dataDateStr;
          })
          .reduce((shiftSum, shift) => shiftSum + shift.payroll, 0);
        return sum + dayPayroll;
      }, 0);

    // Calculate the sum of payroll for monthly and weekly employees
    const nonHourlyPayroll = this.tempEmps
      .filter(emp => emp.employeeType === "MONTHLYEMPLOYEE" || emp.employeeType === "WEEKLYEMPLOYEE")
      .reduce((sum, emp) => sum + emp.payroll, 0);

    payroll = hourlyPayroll + nonHourlyPayroll;
  }

  return payroll;
}

  calculateTotalTimeDifference(data) {
    let totalDifferenceMilliseconds = 0;

    for (const entry of data) {
      const startTime = new Date(entry.startTime);
      const endTime: Date = new Date(entry.endTime); // Explicitly specify the type as Date
      totalDifferenceMilliseconds += endTime.getTime() - startTime.getTime();
    }

    const totalDifferenceHours = Math.floor(totalDifferenceMilliseconds / (1000 * 60 * 60));
    const totalDifferenceMinutes = Math.floor((totalDifferenceMilliseconds % (1000 * 60 * 60)) / (1000 * 60));
    return `${totalDifferenceHours} h ${totalDifferenceMinutes} m`;
  }
