import { DatePipe, formatDate } from "@angular/common";
import { Inject, Injectable, LOCALE_ID } from "@angular/core";
import { locale } from "moment";
import { NGXLogger } from "ngx-logger";

import { BehaviorSubject, Observable, pipe, Subject, Subscriber } from "rxjs";
import { take, takeUntil } from "rxjs/operators";
import { AuthFacadeService } from "../../auth/facade/auth-facade.service";

import { CashUp } from "../model/cashup.model";
import { CashupService } from "../service";

@Injectable()
export class CashupFacadeService {
  private cashupListSubject = new BehaviorSubject<CashUp[]>([]);
  cashupList$ = this.cashupListSubject.asObservable();
  private cashListSubject = new BehaviorSubject<CashUp[]>([]);
  cashList$ = this.cashListSubject.asObservable();
  private empListSubject = new BehaviorSubject<any>([]);
  empList$ = this.empListSubject.asObservable();
  private fetchListSubject = new BehaviorSubject<any>([]);
  fetchList$ = this.fetchListSubject.asObservable();
  AzureEmail = localStorage.getItem("AzureEmail");
  private dataLoaded = new BehaviorSubject<any>(false);
  dataLoaded$ = this.dataLoaded.asObservable();
  private readonly destroy$ = new Subject<void>();
  datePipe = new DatePipe("en-US");

  constructor(
    private cashupDataService: CashupService,
    private authFacade: AuthFacadeService,
    private logger: NGXLogger,
    @Inject(LOCALE_ID) public locale: string
  ) {
    this.setCashupPermissions();
    this.authFacade.dataLoaded$
      .pipe(take(2))
      .subscribe((x) => this.dataLoaded.next(true));
  }

  private dataStore = {
    all_cashup: [],
    all_cashupList: [],
    cashupList: [],
    all_Emp: [],
    fetch: [],
  };
  setCashupPermissions() {
    let user = this.authFacade.getUserDetails();
    let dataload = this.authFacade.getdataLoad();
    this.logger.log("User from cashup fascade", user);
  }
  page:number=0;
  load(): void {
    this.authFacade
      .getUserRestaurants(localStorage.getItem("AzureEmail"))
      .pipe(take(1))
      .subscribe(
        (x: any) => {
          this.cashupDataService
            .getAllCashupByPage(localStorage.getItem("UserRestaurantId"), 10, this.page)
            .pipe()
            .subscribe(
              (data) => {
                let cashupList=data['cashUps'];
                // this.cashupListSubject.next(cashupList);
                if (cashupList.hasOwnProperty("error")) {
                  this.logger.log("No Cashup Exist");
                  this.dataStore.all_cashup = [];
                } else {
                  this.logger.log("cashup data loaded");
                  cashupList.map((x) => {
                    x["difference"] = this.getCashupTotals(x).difference;
                    x.cashUpDate = new Date(x.cashUpDate);
                  });
                  this.dataStore.all_cashup = cashupList;
                }
                console.log("allcashup", this.dataStore.all_cashup);
                return this.cashupListSubject.next(
                  Object.assign({}, this.dataStore).all_cashup
                );
              },
              (err) => {
                this.logger.error("err", err);
              }
            );
        },
        (err) => console.log("error restaurant", err)
      );
  }

getAllCashups(){
  this.cashupDataService
            .getAllPendingDeposit(localStorage.getItem("UserRestaurantId"), 100, this.page)
            .pipe()
            .subscribe(
              (data) => {
                let cashList=data['cashUps'];
                // this.cashupListSubject.next(cashupList);
                if (cashList.hasOwnProperty("error")) {
                  this.logger.log("No Cashup Exist");
                  this.dataStore.all_cashupList = [];
                } else {
                  this.logger.log("cashup data loaded");
                  cashList.map((x) => {
                    x["difference"] = this.getCashupTotals(x).difference;
                    x.cashUpDate = new Date(x.cashUpDate);
                  });
                  this.dataStore.all_cashupList = cashList;
                }
                console.log("allcashup", this.dataStore.all_cashupList);
                return this.cashListSubject.next(
                  Object.assign({}, this.dataStore).all_cashupList
                );
              },
              (err) => {
                this.logger.error("err", err);
              }
            );
}
getAllDrafts(){
  this.cashupDataService
            .getAllDrafts(localStorage.getItem("UserRestaurantId"), 10, this.page)
            .pipe()
            .subscribe(
              (data) => {
                let cashupList=data['cashUps'];
                // this.cashupListSubject.next(cashupList);
                if (cashupList.hasOwnProperty("error")) {
                  this.logger.log("No Cashup Exist");
                  this.dataStore.all_cashup = [];
                } else {

  getCashupTotals(cash: CashUp) {
    let res = {
      epos: 0,
      cash: 0,
      pdq: 0,
      delivery: 0,
      kpi_total: 0,
      difference: 0,
      wages: 0,
      petty: 0,
      third_party: 0,
      covers: 0,
      refund_breakdown: 0,
      discount_breakdown: 0,
    };

    if (!cash || !cash.sales) return res;
    res.epos +=
      cash?.sales[0]?.foodPayment +
      cash?.sales[0]?.drinksPayment +
      cash?.sales[0]?.otherPayment +
      cash?.sales[0]?.serviceCharges -
      cash?.sales[0]?.discount +
      cash?.sales[0]?.creditCardTip +
      cash?.sales[0]?.taxInfo[0]?.amount;

    res.cash += cash?.cashnPdq?.tillSystems?.reduce(
      (sum, x) => sum + x.amount,
      0
    );

    res.pdq += cash?.cashnPdq?.pdqSystems?.reduce(
      (sum, x) => sum + x.amount,
      0
    );
    res.petty += cash?.cashnPdq?.pettyCashs?.reduce(
      (sum, x) => sum + x.amount,
      0
    );
    res.third_party += cash?.thirdPartyInfo?.reduce(
      (sum, x) => sum + x.amount,
      0
    );
    res.covers += cash?.kpi?.kpiCovers
      ?.filter((y) => y.kpiName != "Void")
      .reduce((sum, x) => sum + x.amount, 0);
    res.refund_breakdown += cash?.kpi?.breakDownDetails
      ?.filter((x) => x.name == "refund_breakdown")
      .reduce((sum, y) => sum + y.amount, 0);
    res.discount_breakdown += cash?.kpi?.breakDownDetails
      ?.filter((x) => x.name == "discount_breakdown")
      .reduce((sum, y) => sum + y.amount, 0);
    res.kpi_total += cash?.kpi?.breakDownDetails?.reduce(
      (sum, x) => sum + x.amount,
      0
    );
    res.wages += cash?.cashnPdq?.wageAdvances?.reduce(
      (sum, x) => sum + x.amount,
      0
    );
    // Third Party
    res.delivery += cash?.thirdPartyInfo?.reduce((sum, x) => sum + x.amount, 0);

    //  NOW ----  (Cash + Petty Cash + PDQ + Third Party ) - EPOS Total
    //  Prev ---- EPOS Total - (Cash Total +  PDQ Total + Third Party Total )
    res.difference +=
      res.petty + res.cash + res.pdq + res.delivery + res.wages - res.epos;

    return res;
  }
