import { HttpClient, HttpErrorResponse } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from '../../../../environments/environment';
import All_Request from '../model/request.model';
import { catchError, retry } from 'rxjs/operators';
import { throwError } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class RequestService {

  constructor(private http: HttpClient) { }

  getAllRequest(name, id) {
    const RequestUrl = `${environment.HRUrl}leaveRequest/getAllRequests?username=` + `${name}` + `&restaurantId=` + `${id}`
    return this.http.get<All_Request[]>(RequestUrl);
  }

  getRequestByName(name, id) {
    const RequestUrl = `${environment.HRUrl}leaveRequest/getEmployeeRequests?username=` + `${name}` + `&restaurantId=` + `${id}`
    return this.http.get<All_Request[]>(RequestUrl);
  }


  //bug punch in out
  getAllShiftRequest(name, id) {
    // https://hrservice.test.restaurantonesolution.com/hr/request/shift/shiftRequest/getEmployeeRequests?username=nnnn&restaurantId=4e41e7c6-d38c-4861-acdd-b4806b34c141
    const RequestUrl = `${environment.HRUrl}request/shift/shiftRequest/getAllRequests?username=` + `${name}` + `&restaurantId=` + `${id}`;
    return this.http.get<All_Request[]>(RequestUrl);
  }
  // punch in out
  updatefoodicsData(date, data) {
    let rosClientId = JSON.parse(localStorage.getItem("User")).userProfile.id;
    // https://hrservice.test.restaurantonesolution.com/hr/request/shift/shiftRequest/getEmployeeRequests?username=nnnn&restaurantId=4e41e7c6-d38c-4861-acdd-b4806b34c141
    const RequestUrl = `${environment.HRUrl}shift/foodicsClockInOut?rosClientId=${rosClientId}` + `&shiftDateString=` + `${date}`;
    return this.http.put(RequestUrl,data);
  }

  private handleError(error: HttpErrorResponse) {
    console.error('An error occurred:', error);
    return throwError(error);
  }

  updateEposData(date, data) {
    let rosClientId = localStorage.getItem("UserRestaurantId");
    console.log(rosClientId,"====client id");
    
    // https://hrservice.test.restaurantonesolution.com/hr/request/shift/shiftRequest/getEmployeeRequests?username=nnnn&restaurantId=4e41e7c6-d38c-4861-acdd-b4806b34c141
    const RequestUrl = `${environment.HRUrl}shift/eposClockInOut?rosRestaurantId=${rosClientId}` + `&shiftDateString=` + `${date}`;
    return this.http.put(RequestUrl,data).pipe(
      retry(3), // Retry up to 3 times before failing
      catchError(this.handleError)
    );;
  }

 

  getEmpShiftRequest(name, id) {
    const RequestUrl = `${environment.HRUrl}request/shift/shiftRequest/getEmployeeRequests?username=` + `${name}` + `&restaurantId=` + `${id}`;
    return this.http.get<All_Request[]>(RequestUrl);
  }
  approveRequest(e) {
    const empUrl = `${environment.HRUrl}leaveRequest/editLeaveRequests`;
    return this.http.put<any>(empUrl, e);
  }
  editShiftReq(e) {
    console.log("++++++++++++++++++++++++++++++",JSON.stringify(e));
    const empUrl = `${environment.HRUrl}request/shift/shiftRequest/editShiftRequests`;
    return this.http.put<any>(empUrl, e);
  }

  updateShiftReq(e){
    console.log("++++++++++++++++++++++++++++++",JSON.stringify(e));
    const empUrl = `${environment.HRUrl}shift/editShift`;
    return this.http.put<any>(empUrl, e);
  }

  editShift(e){
    const empUrl = `${environment.HRUrl}shift/editShift`;
    return this.http.put<any>(empUrl, e);
  }

  deleteShiftReq(id) {
    const empUrl = `${environment.HRUrl}request/shift/shiftRequest/deleteShiftRequests?shiftRequestId=${id}`;
    return this.http.delete<any>(empUrl);
  }
  getEmployeeProfile(managerEmail, id) {
    const RequestUrl = `${environment.HRUrl}request/profile/getRequestForReportingEmployee?username=${managerEmail}&restaurantId=${id}`;
    return this.http.get<any>(RequestUrl);
  }
  getMyProfile(managerEmail) {
    const RequestUrl = `${environment.HRUrl}request/profile/getRequestForEmployee?username=${managerEmail}`;
    return this.http.get<any>(RequestUrl);
  }
  deleteMyProfileReq(profileRequestInfoId) {
    const empUrl = `${environment.HRUrl}request/profile/deleteRequest?profileRequestInfoId=${profileRequestInfoId}`;
    return this.http.delete<any>(empUrl);
  }
  editProfileStatus(jsonObject) {
    const empUrl = `${environment.HRUrl}request/profile/editProfileRequestStatus`;
    return this.http.put<any>(empUrl, jsonObject);
  }
}
