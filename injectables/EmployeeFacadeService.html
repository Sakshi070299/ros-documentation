  getEmpShiftCalendar() {
    let ALL_EMP;
    let ALL_SHIFT;
    this.allEmp$ = this.employeeService.getAllEmployee();
    this.allShift$ = this.shiftCalendarService.getAllShifts();
    forkJoin({
      allEmp: this.allEmp$,
      allShift: this.allShift$,
    })
      .pipe(shareReplay())
      .subscribe(({ allEmp, allShift }) => {
        ALL_EMP = allEmp;
        ALL_SHIFT = allShift;
        ALL_EMP.map((e) => {
          let shifts = ALL_SHIFT.filter((s) => s.empId === e.empId);
          e.shifts = shifts;
        });
        this.logger.log("EMPLOYEE SHIFT CALENDAR Data is fetched");
        this.logger.log("EMP", ALL_EMP);
        this.empshiftCalendarSubject$.next(ALL_EMP);
      },
      (err)=>{
        console.log(err,"err");
        
      }
      );
  }

  load(): void {
    this.employeeService
      .getAllEmployee()
      .pipe(take(1))
      .subscribe(
        (empList) => {
          this.logger.log("load data completed");
          this.dataStore.allEmployee = empList;

          // Filtering employees for the selected restaurant
          let restaurantId = localStorage.getItem("UserRestaurantId");
          if (restaurantId) {
            this.dataStore.allEmployee = empList.filter((x) =>
              x.restaurants?.find((y) => y.restaurantId == restaurantId)
            );
          }

          return this.employeeListSubject.next(
            Object.assign({}, this.dataStore).allEmployee
          );
        },
        (err) => {
          this.logger.log("err", err);
        }
      );
  }

  editEmployee(e: any): Observable<any>{
    return new Observable((Subscriber) => {

      .editEmployee(e)

      .subscribe(
        (e) => {
          this.logger.log(e);
          for (let i = 0; i < this.dataStore.allEmployee.length; i++) {
            if (this.dataStore.allEmployee[i].id === e.empId) {
              this.dataStore.allEmployee[i] = e;
              this.logger.log(
                "editEmployee data",

              );
            }
          }
          this.logger.log("datastore", this.dataStore.allEmployee);

            Object.assign({}, this.dataStore).allEmployee
          );
          Subscriber.next(e);
            Subscriber.complete();
        },
        (err) => this.logger.log(err)
      );
    })
  }

  getEmployeeProfile(id: string) {
    this.logger.log("Inside getEMPLOYEEPROFILE", id);
    let EMP;
    let SHIFT;
    this.empProfile$ = this.employeeService.getEmployById(id);
    this.empShiftsProfile$ =
      this.shiftCalendarService.getShiftByEmployeeById(id);

    forkJoin({
      emp: this.empProfile$,
      shift: this.empShiftsProfile$,
    })
      .pipe(shareReplay())
      .subscribe(({ emp, shift }) => {
        EMP = emp;
        SHIFT = shift;
        EMP.shifts = SHIFT;
        this.logger.log("EMPLOYEE PROFILE loaded");
        this.logger.log("EMP PROFILE: ", EMP);
        this.empProfileSubject.next(EMP);
      },(err)=>{
        this.logger.log(err)
      });
