          (recList) => {
            // this.cashupListSubject.next(cashupList);

              this.dataStore.reconcile_Summary = [];

              console.log("rec data loaded");
              this.dataStore.reconcile_Summary = recList;

            return this.reconcilationSummaryListSubject.next(

            );
          },
          (err) => {
            this.logger.error("err", err);
          }
         );
        } , (err)=>console.log("error restaurant",err))

      }
      getReconciledSummary(start,end):void{

        .getAllReconciledSummary(start,end,localStorage.getItem("UserRestaurantId"))
        .pipe()
        .subscribe(

           }
           return this.reconciledSummaryListSubject.next(
             Object.assign({}, this.dataStore).reconciled_Summary
           );
         },
         (err) => {
           this.logger.error("err", err);
         }
        );
      } , (err)=>console.log("error restaurant",err))
     }
