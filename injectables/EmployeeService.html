import { HttpClient, HttpHeaders, HttpParams } from "@angular/common/http";
import { Injectable } from "@angular/core";
import { Observable } from "rxjs";
import { environment } from "../../../../environments/environment";
import { Employee } from "../model/employee.model";
import { RestaurantShiftConfig } from "../model/resturantShiftConfig.model";

@Injectable()
export class EmployeeService {
  constructor(private http: HttpClient) {}

  // shiftPublishedStatus(restaurantId, startDate, endDate) {
  //   const body = {}
  //   const empUrl = `${environment.HRUrl}shift/publishedStatus?restaurantId=${restaurantId}&startDate=${startDate}&endDate=${endDate}`;
  //   return this.http.post<any>(empUrl, body);
  // }

 

  getAllUpcomingShifts(id,date){
    const empUrl = `${environment.HRUrl}shift/GetAllUpcomingShifts?empId=${id}&currentDate=${date}`;
    return this.http.get<any>(empUrl)
  }

  getAllUpcomingLeaves(id,date){
    const empUrl = `${environment.HRUrl}/GetAllFutureLeaveRequests?empId=${id}&currentDate=${date}`;
    return this.http.get<any>(empUrl)
  }

  getAllReporrtingEmployees(username,restId){
    const empUrl = `${environment.HRUrl}employee/getAllReportingEmployees?username=${username}&restaurantId=${restId}`;
    return this.http.get<any>(empUrl)
  }
  //  To transfer the reporting employee name
  transferReportingEmployee(restId,fromusername,tousername){
    const empUrl = `${environment.HRUrl}employee/transfer?restaurantId=${restId}&employeeFromUsername=${fromusername}&employeeToUsername=${tousername}`;
    return this.http.patch(empUrl, {});
  }

  // To Call the deactivate employee method
  deactivateEmployee(username,status){
    const empUrl = `${environment.HRUrl}employee/deActivateEmployee?employeeUsername=${username}&status=${status}`;
    return this.http.patch(empUrl, {});
  }


  deleteShifts(id,date){
     const empUrl = `${environment.HRUrl}shift/DeleteAllUpcomingShifts?empId=${id}&currentDate=${date}`;
     return this.http.delete<any>(empUrl);
  }


  deleteLeaves(id,date){
    const empUrl = `${environment.HRUrl}DeleteAllFutureLeaveRequests?empId=${id}&currentDate=${date}`;
    return this.http.delete<any>(empUrl);
  }

  getAllEmployee() {
    const empUrl = `${environment.HRUrl}employee`;
    return this.http.get<any>(empUrl);
  }

  addEmployee(emp: any) {
    const empUrl = `${environment.HRUrl}employee`;
    return this.http.post<any>(empUrl, emp);
  }

  getAllReportingEmployee(
    username: string,
    restaurantId: string
  ): Observable<any> {
    const empUrl = `${environment.HRUrl}employee/getAllReportingEmployees?username=${username}&restaurantId=${restaurantId}`;
    //  const params = new HttpParams().set("username", username).set("restaurantId", restaurantId);
    return this.http.get<any>(empUrl);
  }
  editEmployee(emp: any) {
    const empUrl = `${environment.HRUrl}request/profile/addProfileRequest`;
    return this.http.post<any>(empUrl, emp);
  }
