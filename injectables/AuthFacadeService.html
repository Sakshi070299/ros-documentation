                // "acr" claim in the token tells us what policy is used (NOTE: for new policies (v2.0), use "tfp" instead of "acr")
                // To learn more about b2c tokens, visit https://docs.microsoft.com/en-us/azure/active-directory-b2c/tokens-overview
                if (
                  success.idToken.claims.acr === b2cPolicies.names.resetPassword
                ) {
                  window.alert(
                    "Password has been reset successfully. \nPlease sign-in with your new password"
                  );
                  return this.authService.logout();

                alert(
                  "User is Deactived ! Kindly activate your account to Login"
                );
                this.logout();
              }
            } else {
              console.log("Couldn't access Userinfo form api");
            }
          });
      }
    );

    loginFailureSubscription = this.broadcastService.subscribe(
      "msal:loginFailure",
      (error) => {
        console.log("login failed");
        console.log(error);

        if (error.errorMessage) {
          // Check for forgot password error
          // Learn more about AAD error codes at https://docs.microsoft.com/en-us/azure/active-directory/develop/reference-aadsts-error-codes
          if (error.errorMessage.indexOf("AADB2C90118") > -1) {
            if (isIE) {
              this.msalService.loginRedirect(
                b2cPolicies.authorities.resetPassword
              );
            } else {
              this.msalService.loginRedirect(
                b2cPolicies.authorities.resetPassword
              );
            }
          }
        }
      }
    );

    // redirect callback for redirect flow (IE)
    this.msalService.handleRedirectCallback((authError, response) => {
      if (authError) {
        //this.login("ros@gmail.com", 12345);
        console.error("Redirect Error: ", authError.errorMessage);
        return;
      }

    });

      new Logger(
        (logLevel, message, piiEnabled) => {
        },
        {
          correlationId: CryptoUtils.createNewGuid(),
          piiLoggingEnabled: false,
        }
      )
    );

    this.subscriptions.push(loginSuccessSubscription);
    this.subscriptions.push(loginFailureSubscription);
  }

  configuration() {
    this.restaurantIdSubject.subscribe((x) => {
      // console.log("congifuguration", x)
      if (x) {
        // console.log("ifcondition", x)

  getRestaurantID() {
    this.UserLoginSubject.subscribe((x) => {
      if (x) {
        this.permissionList$ = this.getUserPermissions(x);
        this.permissionList$.subscribe((y) => {
          let filter = y.userPermissions.filter((z) => z.permitted == true);
          this.UserPermissions = filter.map((a) => a.featureCode);
          localStorage.setItem(
            "Permissions",
            JSON.stringify(this.UserPermissions)
          );
          this.ngxPermissions.loadPermissions(
            JSON.parse(localStorage.getItem("Permissions"))
          );
        });
      }
    });
  }
  getUserRestaurants(id) {
    return new Observable((Subscriber) => {
      this.authService
        .getUserRestaurants(id)
        .pipe(takeUntil(this.destroy$))
        .subscribe(
          (resList) => {
            if (!resList || resList.hasOwnProperty("error")) {
              console.log("can't get userRestaurant");
              this.dataStore.all_Restaurant = [];
            } else {
              //  console.log("res data loaded");
              this.dataStore.all_Restaurant = resList;
              this.selectedRestaurants = resList.restaurants[0];
              if (!localStorage.getItem("UserRestaurant")) {
                localStorage.setItem(
                  "UserRestaurant",
                  JSON.stringify(resList.restaurants[0])
                );
              }

              this.restaurantIdSubject.next(resList.restaurants[0].id);
              if (!localStorage.getItem("UserRestaurantId")) {
                localStorage.setItem(
                  "UserRestaurantId",
                  resList.restaurants[0].id
                );
              }

              this.ResListSubject.next(
                Object.assign({}, this.dataStore).all_Restaurant
              );
            }
            Subscriber.next(resList);
            Subscriber.complete();
          },
          (err) => Subscriber.error(err)
        );
    });
  }
