import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable, Subject } from 'rxjs';
import { endWith } from 'rxjs/operators';
import { environment } from '../../../../environments/environment';
import { CashUp } from '../model/cashup.model';
import { reconcilationSummary } from '../model/reconcilation.model';
import {ReconcileLandingComponent} from '../reconciliation/reconcile-landing/reconcile-landing.component'

@Injectable({
  providedIn: 'root',
})
export class ReconciliationService {
  constructor( private http : HttpClient) { 
  }
  
getAllReconciles(){
  return this.reconcile;
}
getAllReconcile(){
  const url = `${environment.backendUrl}` + "";
  return this.http.get<any>(url);
}
getAllReconcileSummary(start_date,end_date,id){
  const url = `${environment.backendUrl}account/reconciliation/summary?fromDate=${start_date}&toDate=${end_date}&resturantID=${id}`;
  return this.http.get<reconcilationSummary[]>(url);
}
getAllReconciledSummary(start_date,end_date,id){
  const url = `${environment.backendUrl}account/reconciliation/reconciledsummary?fromDate=${start_date}&toDate=${end_date}&restuarantId=${id}`;
  return this.http.get<reconcilationSummary[]>(url);
}
getAllReconcilationCalendar(start_date,end_date,activecard,id){
  const url = `${environment.backendUrl}account/reconciliation/reconciledcalendarsummary?fromDate=${start_date}&toDate=${end_date}&restuarantId=${id}&rType=${activecard}`;
  return this.http.get<any>(url);
}
getCardSheets(id){
  const url = `${environment.backendUrl}account/reconciliation/getsheetsforcard?id=${id}`;
  return this.http.get<any[]>(url);
}
getThirdpartySheets(id){
  const url = `${environment.backendUrl}account/reconciliation/getsheetsforthirdparty?id=${id}`;
  return this.http.get<any[]>(url);
}
getCashSheets(id){
  const url = `${environment.backendUrl}account/reconciliation/getsheetsforcash?id=${id}`;
  return this.http.get<any[]>(url);
}
cardsReconciled(cards){
  const url = `${environment.backendUrl}account/reconciliation/cardReconciliation`;
  return this.http.post<any[]>(url,cards);
}
thirdpartyReconciled(thirdparty){
  const url = `${environment.backendUrl}account/reconciliation/thirdPartyReconciliation`;
  return this.http.post<any[]>(url,thirdparty);
}
cashReconciled(cash){
  const url = `${environment.backendUrl}account/reconciliation/cashReconciliation`;
  return this.http.post<any[]>(url,cash);
}

getBankingByDate(date:string){
  const url = `${environment.backendUrl}account/cashup/bankingByDate?bankingDate=${date}&restaurantId=${localStorage.getItem("UserRestaurantId")}`;
  return this.http.get<any[]>(url);
}
